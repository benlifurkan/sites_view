    int verilecek = 0, miktar = 0, para200 = 0, para100 = 0, para50 = 0, para20 = 0, para10 = 0, para5 = 0;
    int banka200 = 10, banka100 = 4, banka50 = 5, banka20 = 20, banka10 = 10, banka5 = 1000;
    
    
    Console.WriteLine("Merhabalar");
    Console.WriteLine("YALÇINBANK a hoş geldiniz.");
    Console.WriteLine("--------------------------");
    Console.WriteLine("Bankamızdan 5 ve 5 in katı para birimi çekebilirsiniz.");
    
                        
    int butce = banka200 * 200 + banka100 * 100 + banka50 * 50 + banka20 * 20 + banka10 * 10 + banka5 * 5;
    
                        
    do
    {
        Console.Write("Lütfen miktarı giriniz.\nMiktar: ");
        miktar = Convert.ToInt32(Console.ReadLine());
        if (butce < miktar)
            Console.WriteLine("Paramız bitti. Daha düşük bir miktar giriniz.");
    } while (miktar < 5 || miktar % 5 != 0 || butce < miktar);
    
    verilecek = miktar;
                        
    do
    {
        if (miktar >= 200 && banka200 > 0)
        {
            para200++;
            banka200--;
            miktar -= 200;
        }
        else if (miktar >= 100 && banka100 > 0)
        {
            para100++;
            banka100--;
            miktar -= 100;
        }
        else if (miktar >= 50 && banka50 > 0)
        {
            para50++;
            banka50--;
            miktar -= 50;
        }
        else if (miktar >= 20 && banka20 > 0)
        {
            para20++;
            banka20--;
            miktar -= 20;
        }
        else if (miktar >= 10 && banka10 > 0)
        {
            para10++;
            banka10--;
            miktar -= 10;
        }
        else if (miktar >= 5 && banka5 > 0)
        {
            para5++;
            banka5--;
            miktar -= 5;
        }
                        
    } while (miktar >= 5);
                        
                        
    Console.WriteLine("Talep ettiniz : " + verilecek + "TL");
    if (para200 > 0)
        Console.WriteLine("200TL x " + para200);
    if (para100 > 0)
        Console.WriteLine("100TL x " + para100);
    if (para50 > 0)
        Console.WriteLine("50TL x " + para50);
    if (para20 > 0)
        Console.WriteLine("20TL x " + para20);
    if (para10 > 0)
        Console.WriteLine("10TL x " + para10);
    if (para5 > 0)
        Console.WriteLine("5TL x " + para5);
Programda kullanılanacak değişkenler tanımlandı ve bankadaki para miktarları belirlendi.
Kullanıcı karşılandı, kullanıcıya 5 ve 5 in katı olacak şekilde miktar girmesi gerektiğinin bilgisi verildi.
Kullanıcıdan alınan miktarın 5 ve 5 in katı olup olmadığı kontrol edildi. Eğer 5 ve 5 in katı değilse ve bankadaki para miktarı kullanıcıdan alınan miktarı karşılayamıyorsa kullanıcıdan tekrar miktar istendi.
Kullanıcıdan alınan miktar bankadaki para miktarı ile karşılanabiliyorsa kullanıcıya verilecek para miktarı hesaplandı.
Kullanıcıya verilecek para miktarı ekrana yazdırıldı.
